import responses

@responses.activate
def test_place_order_rate_limit():
    # 第一次 429，第二次成功
    responses.add(responses.POST, "https://paper-api.alpaca.markets/v2/orders",
                  status=429, json={"message": "too many requests"})
    responses.add(responses.POST, "https://paper-api.alpaca.markets/v2/orders",
                  status=200, json={"id": "abc", "status": "new"})

    client = AlpacaClient(spec, env)
    result = client.place_order("AAPL", qty=1, side="buy")
    assert result.order_id == "abc"
    assert len(responses.calls) == 2

def test_secrets_never_in_logs(caplog):
    with caplog.at_level(logging.DEBUG):
        client = AlpacaClient(spec={"id":"alpaca", ...},
                              env={"ACC1_API_SECRET": "SUPER_SECRET_xyz"})
        try:
            client.get_account_balance()
        except:
            pass

    for record in caplog.records:
        assert "SUPER_SECRET_xyz" not in record.message
        assert "SUPER_SECRET_xyz" not in str(record.args)

🔙 Rollback 計劃

任何時候發現問題可立即回到舊版：

# 1. 立刻停所有 workflow（防止繼續下單）
gh workflow disable daily_all_accounts.yml

# 2. 取消所有 open orders
python3 -c "from trader import client_from_env; client_from_env().cancel_all_open_orders()"

# 3. revert 到上一個 known-good commit
git revert HEAD~N..HEAD  # 或
git reset --hard <last-good-sha>
git push --force-with-lease

# 4. 重新啟用 workflow
gh workflow enable daily_all_accounts.yml

# 5. 手動觸發 dry-run 驗證舊版正常
gh workflow run daily_all_accounts.yml -f dry_run=true
